# Checkbox

Form Checkbox component. Different sizes are available.

## Props

<PropsTable of={Checkbox} />

## Examples

### Default Checkbox
<Playground>
  <Checkbox id="checkbox" name="checkbox" />
</Playground>

### Checkbox Group vertical
<Playground>
  {() => {
    const checkboxValues = [
      {
        id: 1,
        value: 'test',
        label: 'Checkbox 1',
      },
      {
        id: 2,
        value: 'test2',
        label: 'Checkbox 2',
      },
    ];

    return (
      <div>
        <CheckboxGroup name="checkboxes" id="checkboxes" choices={checkboxValues} />
      </div>
    )
  }}
</Playground>

### Checkbox Group exclusivity
<Playground>
  {() => {
    const checkboxValues = [
      {
        id: 1,
        value: 'test',
        label: 'Checkbox 1',
      },
      {
        id: 2,
        value: 'test2',
        label: 'Checkbox 2',
      },
      {
        id: 3,
        value: 'neither',
        label: 'Neither',
        exclusive: true,
      },
    ];

    return (
      <div>
        <CheckboxGroup horizontal name="checkboxes" id="checkboxes" choices={checkboxValues} />
      </div>
    )
  }}
</Playground>
